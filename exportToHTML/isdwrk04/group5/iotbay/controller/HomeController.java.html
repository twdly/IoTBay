3    import javax.servlet.ServletException;
4    import javax.servlet.annotation.WebServlet;

9    
10   import isdwrk04.group5.iotbay.dao.ProductDao;
11   import isdwrk04.group5.iotbay.model.Cart;
12   import isdwrk04.group5.iotbay.model.Product;
13   
14   
15   @WebServlet(name = "homeController", value = {"", "/search"})
16   public class HomeController extends BaseServlet {
17   
18       public void init() {
19       }
20   
21       @Override
22       public void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
23           String query = request.getParameter("search");
24           String category = request.getParameter("category");
25   
26           ProductDao dao = new ProductDao();
27           List<Product> products;
28           List<String> productCategories = dao.getProductCategories();
29   
30           if (query != null && !query.isEmpty()) {
31               products = dao.getSearchProducts(query);
32               request.setAttribute("searchQuery", query);
33           } else if (category != null && !category.isEmpty()) {
34               products = dao.getCategoryProducts(category);
35               request.setAttribute("category", category);
36           } else {
37               products = dao.getAllProducts();
38           }
39   
40           if (request.getAttribute("cart") == null) {
41               request.setAttribute("cart", new Cart());
42           }
43   
44           request.setAttribute("products", products);
45           request.getSession().setAttribute("productCategories", productCategories);
46   
47           serveJSP(request, response, "home.jsp");
48       }
49   
50       @Override
51       public void doPost(HttpServletRequest request, HttpServletResponse response) {
52           // This method is used to add items to the cart
53           int itemId = Integer.parseInt(request.getParameter("itemId"));
54           Cart cart = (Cart) request.getSession().getAttribute("cart");
55   
56           if (cart == null) {
57               cart = new Cart();
58           }
59   
60           ProductDao productDao = new ProductDao();
61           cart.addItem( productDao.getProductById(itemId));
62   
63           request.getSession().setAttribute("cart", cart);
64   
65           // This is necessary to redirect the user back to the current page they are on
66           // Doing so allows all their currently viewed products to still display as if they were never redirected
67           redirectToUrl(request, response, getRedirectUrl(request));
68       }
69   
70       private static String getRedirectUrl(HttpServletRequest request) {
71           String requestUrl = String.valueOf(request.getRequestURL());
72           String contextPath = request.getContextPath();
73           String query = request.getQueryString();
74           return requestUrl.substring(requestUrl.indexOf(contextPath) + contextPath.length()) + (query != null ? "?" + query : "");
75       }
