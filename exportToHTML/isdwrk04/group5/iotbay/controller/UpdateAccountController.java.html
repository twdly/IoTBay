3    import isdwrk04.group5.iotbay.dao.UserDao;

7    import javax.servlet.ServletException;
8    import javax.servlet.annotation.WebServlet;
9    import javax.servlet.http.HttpServletRequest;
10   import javax.servlet.http.HttpServletResponse;
11   import javax.servlet.http.HttpSession;
12   import java.io.IOException;
13   import java.nio.charset.StandardCharsets;
14   import java.security.NoSuchAlgorithmException;
15   import java.security.spec.InvalidKeySpecException;
16   import java.util.ArrayList;
17   import java.util.List;
18   import java.util.Collections;
19   
20   @WebServlet( name= "updateAccountController", value = "/update-account" )
21   public class UpdateAccountController extends BaseServlet {
22   
23       private HashingService hashingService;
24       private UserDao userDao;
25   
26       @Override
27       public void init() {
28           this.userDao = new UserDao();
29           try {
30               hashingService = new HashingService();
31           } catch (NoSuchAlgorithmException e) {
32               throw new RuntimeException(e);
33           }
34       }
35   
36       @Override
37       public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
38           User user = (User) request.getSession().getAttribute("user");
39   
40           if (null == user) {
41               redirectToUrl(request, response, "/login");
42               return;
43           }
44           serveJSP(request, response, "updateAccount.jsp");
45       }
46   
47   
48       @Override
49       public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
50           HttpSession session = request.getSession();
51           User user = (User) session.getAttribute("user");
52   
53           String action = request.getParameter("action");
54   
55           if ("updateDetails".equals(action)) {
56               updateDetails(request, response, user, session);
57           } else if ("updatePassword".equals(action)) {
58               updatePassword(request, response, user, session);
59           } else if ("deactivate".equals(action)) {
60               deactivateAccount(request, response, user, session);
61           } else {
62               session.setAttribute("errors", Collections.singletonList("Invalid action."));
63               serveJSP(request, response, "updateAccount.jsp");
64           }
65       }
66   
67       public void updateDetails(HttpServletRequest request, HttpServletResponse response, User user, HttpSession session) throws ServletException, IOException {
68   
69           String email = request.getParameter("email");
70           String name = request.getParameter("firstname") + " " + request.getParameter("lastname");
71           String phone = request.getParameter("phoneNumber");
72   
73           if (validateDetails(session, email, name, phone)) {
74               user.setEmail(email);
75               user.setUsername(name);
76               user.setPhoneNo(phone);
77               userDao.updateUserDetails(user);
78               serveJSP(request, response, "updateAccount.jsp");
79           } else {
80               serveJSP(request, response, "updateAccount.jsp");
81           }
82       }
83   
84       public void updatePassword(HttpServletRequest request, HttpServletResponse response, User user, HttpSession session) throws ServletException, IOException {
85           String password = request.getParameter("password");
86           String passwordCheck = request.getParameter("passwordCheck");
87   
88           if (validatePassword(session, password, passwordCheck)) {
89               byte[] salt = hashingService.createSalt();
90               byte[] hashedPassword;
91   
92               try {
93                   hashedPassword = hashingService.hashPassword(salt, password);
94               } catch (InvalidKeySpecException e) {
95                   throw new RuntimeException(e);
96               }
97   
98               user.setHashedPassword(hashedPassword);
99               user.setSalt(salt);
100              userDao.updateUserDetails(user);
101              serveJSP(request, response, "updateAccount.jsp");
102          } else {
103              serveJSP(request, response, "updateAccount.jsp");
104          }
105      }
106  
107      public void deactivateAccount(HttpServletRequest request, HttpServletResponse response, User user, HttpSession session) {
108          String str = "deleted";
109          byte[] byteArray = str.getBytes(StandardCharsets.UTF_8);
110          user.setEmail(str);
111          user.setUsername(str);
112          user.setPhoneNo(str);
113          user.setHashedPassword(byteArray);
114          user.setSalt(byteArray);
115  
116          userDao.updateUserDetails(user);
117          request.getSession(false).invalidate();
118  
119          try {
120              response.sendRedirect(request.getContextPath());
121          } catch (IOException e) {
122              throw new RuntimeException(e);
123          }
124  
125      }
126  
127      private boolean validateDetails(HttpSession session, String email, String name, String phone) {
128          boolean isValid = true;
129          List<String> errors = new ArrayList<>();
130          if (!email.contains("@")) {
131              errors.add("Email is not valid");
132              isValid = false;
133          }
134          if (!phone.matches("\\d{10}") || phone.length() != 10) {
135              errors.add("Phone number can only contain 10-digit number.");
136              isValid = false;
137          }
138          if (!isValid) {
139              session.setAttribute("errors", errors);
140          }
141          return isValid;
142      }
143  
144      private boolean validatePassword(HttpSession session, String password, String passwordCheck) {
145          boolean isValid = true;
146          List<String> errors = new ArrayList<>();
147          if (!password.equals(passwordCheck)) {
148              errors.add("Passwords do not match");
149              isValid = false;
150          }
151          if (password.length() < 8) {
152              errors.add("Password is less than 8 characters");
153              isValid = false;
154          }
155          if (!isValid) {
156              session.setAttribute("errors", errors);
157          }
158          return isValid;
159      }
