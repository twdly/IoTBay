3    import javax.servlet.ServletException;
4    import javax.servlet.http.HttpServlet;
5    import javax.servlet.http.HttpServletRequest;
6    import javax.servlet.http.HttpServletResponse;
7    import java.io.IOException;
8    
9    public abstract class BaseServlet extends HttpServlet {
10   
11       public final String USER_ATTRIBUTE = "user";
12   
13       /**
14        * Method used to redirect the user to the given jsp without updating the URL
15        * Use redirectToUrl instead if you expect the controller method for the page to run
16        * @param jspName the name of the jsp file to redirect to
17        */
18       protected void serveJSP(HttpServletRequest request, HttpServletResponse response, String jspName) throws ServletException, IOException {
19           request.getRequestDispatcher("WEB-INF/" + jspName).include(request, response);
20       }
21   
22       /**
23        * Method used to redirect the user to a given URL.
24        * This allows the user to be redirected while running the controller for the given page
25        * @param url The URL of the controller to redirect to (with leading /)
26        */
27       protected void redirectToUrl(HttpServletRequest request, HttpServletResponse response, String url) {
28           try {
29               response.sendRedirect(request.getContextPath() + url);
30           } catch (IOException e) {
31               throw new RuntimeException(e);
32           }
33       }
34   }
