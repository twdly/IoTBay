1    package isdwrk04.group5.iotbay.controller;

3    import isdwrk04.group5.iotbay.dao.CollectionPointDao;

8    import javax.servlet.ServletException;
9    import javax.servlet.annotation.WebServlet;
10   import javax.servlet.http.HttpServlet;
11   import javax.servlet.http.HttpServletRequest;
12   import javax.servlet.http.HttpServletResponse;
13   import java.io.IOException;
14   
15   @WebServlet(name = "ViewShipmentController", value = "/view-shipment")
16   public class ViewShipmentController extends HttpServlet {
17   
18       private CollectionPointDao collectionPointDao;
19       private DeliveryAddressDao deliveryAddressDao;
20   
21       @Override
22       public void init() {
23           collectionPointDao = new CollectionPointDao();
24           deliveryAddressDao = new DeliveryAddressDao();
25       }
26   
27       @Override
28       protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
29           String cpIdStr = request.getParameter("cpId");
30           String daIdStr = request.getParameter("daId");
31   
32           if (cpIdStr == null || daIdStr == null) {
33               response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Missing parameters.");
34               return;
35           }
36   
37           try {
38               int cpId = Integer.parseInt(cpIdStr);
39               int daId = Integer.parseInt(daIdStr);
40   
41               CollectionPoint collectionPoint = collectionPointDao.getCollectionPointById(cpId);
42               DeliveryAddress deliveryAddress = deliveryAddressDao.getDeliveryAddressById(daId);
43   
44               if (collectionPoint == null || deliveryAddress == null) {
45                   response.sendError(HttpServletResponse.SC_NOT_FOUND, "Shipment not found.");
46                   return;
47               }
48   
49               request.setAttribute("collectionPoint", collectionPoint);
50               request.setAttribute("deliveryAddress", deliveryAddress);
51   
52               request.getRequestDispatcher("viewShipment.jsp").forward(request, response);
53           } catch (NumberFormatException e) {
54               response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Invalid parameters.");
55           }
56       }
57   }
