11   import java.io.IOException;

15   @WebServlet( name = "confirmOrderController", value = "/confirm-order")
16   public class ConfirmOrderController extends BaseServlet {
17   
18       @Override
19       public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
20           Order order = (Order) request.getSession().getAttribute("currentOrder");
21           if (null == order) {
22               redirectToUrl(request, response, "/cart");
23           } else {
24               serveJSP(request, response, "confirmOrder.jsp");
25           }
26       }
27   
28       @Override
29       public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
30           Cart cart = (Cart) request.getSession().getAttribute("cart");
31           Order order = (Order) request.getSession().getAttribute("currentOrder");
32           List<OrderLine> orderLineList = new ArrayList<>();
33           for (Product product : cart.getProducts()) {
34               // The orderID is not yet known hence -1 is used as an obviously wrong placeholder value
35               orderLineList.add(new OrderLine(product.getQuantity(), -1, product.getId()));
36           }
37           OrderDao orderDao = new OrderDao();
38           orderDao.placeOrder(order, orderLineList);
39           ProductDao productDao = new ProductDao();
40           productDao.reduceStockForOrder(orderLineList);
41           request.getSession().removeAttribute("cart");
42           request.getSession().removeAttribute("currentOrder");
43           serveJSP(request, response, "postOrder.jsp");
44       }
45   }
