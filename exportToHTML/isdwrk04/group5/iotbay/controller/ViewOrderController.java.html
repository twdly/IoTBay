1    package isdwrk04.group5.iotbay.controller;
2    
3    import isdwrk04.group5.iotbay.dao.OrderDao;
4    import isdwrk04.group5.iotbay.dao.ProductDao;
5    import isdwrk04.group5.iotbay.model.Order;
6    import isdwrk04.group5.iotbay.model.Product;
7    import isdwrk04.group5.iotbay.model.User;
8    
9    import javax.servlet.ServletException;
10   import javax.servlet.annotation.WebServlet;
11   import javax.servlet.http.HttpServletRequest;
12   import javax.servlet.http.HttpServletResponse;
13   import java.io.IOException;
14   import java.util.List;
15   
16   @WebServlet( name = "viewOrderController", value = "/view-order")
17   public class ViewOrderController extends BaseServlet {
18   
19       public final String CANCEL_TYPE = "cancel";
20       public final String LOGIN_VIEW = "login.jsp";
21       public final String ORDER_VIEW = "viewOrder.jsp";
22   
23       @Override
24       public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
25           User user = (User) request.getSession().getAttribute(USER_ATTRIBUTE);
26           if (null == user) {
27               serveJSP(request, response, LOGIN_VIEW);
28               return;
29           }
30           setOrderAttributes(request, response, user, Integer.parseInt(request.getQueryString()), new OrderDao());
31           serveJSP(request, response, ORDER_VIEW);
32       }
33   
34       @Override
35       public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
36           String requestType = request.getParameter("type");
37   
38           if (requestType.equals(CANCEL_TYPE)) {
39               int orderId;
40               try {
41                   orderId = Integer.parseInt(request.getParameter("number"));
42               } catch (NumberFormatException e) {
43                   response.sendError(400);
44                   return;
45               }
46   
47               OrderDao dao = new OrderDao();
48               Order orderToCancel = dao.getOrderById(orderId);
49               User user = (User) request.getSession().getAttribute(USER_ATTRIBUTE);
50               if (orderToCancel == null || orderToCancel.getUserId() != user.getId()) {
51                   response.sendError(400);
52                   return;
53               }
54   
55               dao.cancelOrder(orderId);
56               setOrderAttributes(request, response, user, orderId, dao);
57   
58               serveJSP(request, response, ORDER_VIEW);
59           }
60       }
61   
62       private void setOrderAttributes(HttpServletRequest request, HttpServletResponse response, User user, int orderId, OrderDao orderDao) throws IOException {
63           ProductDao productDao = new ProductDao();
64   
65           Order order = orderDao.getOrderById(orderId);
66           if (null == order) {
67               response.sendError(404, "Order does not exist");
68               return;
69           }
70   
71           if (order.getUserId() != user.getId()) {
72               response.sendError(403);
73               return;
74           }
75   
76           List<Product> products = productDao.getProductsFromOrder(order.getId());
77           request.setAttribute("order", order);
78           request.setAttribute("products", products);
79       }
80   }
