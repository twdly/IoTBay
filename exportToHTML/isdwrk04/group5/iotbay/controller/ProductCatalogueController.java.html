9    import javax.servlet.http.HttpServletRequest;
10   import javax.servlet.http.HttpServletResponse;
11   import javax.servlet.http.HttpSession;
12   import java.io.IOException;
13   import java.util.ArrayList;
14   import java.util.List;
15   
16   @WebServlet( name="productCatalogueController", value = "/productCatalogue")
17   public class ProductCatalogueController extends BaseServlet {
18   
19       @Override
20       public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
21           User user = (User) request.getSession().getAttribute("user");
22   
23           if (null == user) {
24               redirectToUrl(request, response, "/login");
25               return;
26           }
27   
28           if (!user.getRole().equals(User.Role.Staff)) {
29               redirectToUrl(request, response, "/");
30               return;
31           }
32   
33           ProductDao dao = new ProductDao();
34           List<Product> products;
35           products = dao.getAllProducts();
36   
37           request.setAttribute("products", products);
38   
39           serveJSP(request, response, "productCatalogue.jsp");
40       }
41   
42       @Override
43       public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
44           ProductDao productDao = new ProductDao();
45   
46           HttpSession session = request.getSession();
47   
48           String productName = request.getParameter("product-name");
49           String productCategory = request.getParameter("product-category");
50           String productDescription = request.getParameter("product-description");
51           String productPriceStr = request.getParameter("product-price");
52           String productStockStr = request.getParameter("product-stock");
53   
54           if (validateNewProduct(session, productName, productCategory, productDescription, productPriceStr, productStockStr)) {
55               double productPrice = Double.parseDouble(productPriceStr);
56               int productStock = Integer.parseInt(productStockStr);
57               Product product = new Product(productName, productCategory, productDescription, productPrice, productStock);
58               productDao.addProduct(product);
59           }
60           List<Product> products = productDao.getAllProducts();;
61           request.setAttribute("products", products);
62           serveJSP(request, response, "productCatalogue.jsp");
63       }
64   
65       private boolean validateNewProduct(HttpSession session, String productName, String productCategory, String productDescription, String productPrice, String productStock) {
66           boolean isValid = true;
67           List<String> errors = new ArrayList<>();
68           if (!productName.matches("^[a-zA-Z0-9()\\-\\s]+$")) {
69               errors.add("Product name can only include letters, numbers and the symbols ( ) and -");
70               isValid = false;
71           }
72           if (!productCategory.matches("^[a-zA-Z\\s]+$")) {
73               errors.add("Product category can only contain letters");
74               isValid = false;
75           }
76           if (!productDescription.matches("^[a-zA-Z0-9,.\\s'-]{1,200}$")) {
77               errors.add("Product description can only include letters numbers , and . and must be under 200 characters");
78               isValid = false;
79           }
80           if (!productPrice.matches("^[0-9]{1,5}(?:\\.[0-9]{2})?$")) {
81               errors.add("Product price can only contain 1-5 numbers before the decimal - 00000.00");
82               isValid = false;
83           }
84           if (!productStock.matches("^[0-9]+$")) {
85               errors.add("Stock must be a number");
86               isValid = false;
87           }
88   
89           if (!isValid) {
90               session.setAttribute("errors", errors);
91           }
92           return isValid;
93       }
