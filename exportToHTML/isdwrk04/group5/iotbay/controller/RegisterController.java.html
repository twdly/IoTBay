11   import javax.servlet.http.HttpServletRequest;
12   import javax.servlet.http.HttpServletResponse;

14   import java.io.IOException;
15   import java.security.NoSuchAlgorithmException;
16   import java.security.spec.InvalidKeySpecException;
17   import java.util.ArrayList;
18   import java.util.List;

20   @WebServlet(name = "registerController", value = "/register")
21   public class RegisterController extends BaseServlet {
22   
23       private HashingService hashingService;
24       private UserDao userDao;
25       private AccessLogDAO logDao;
26   
27       @Override
28       public void init() {
29           this.userDao = new UserDao();
30           this.logDao = new AccessLogDAO();
31           try {
32               hashingService = new HashingService();
33           } catch (NoSuchAlgorithmException e) {
34               throw new RuntimeException(e);
35           }
36       }
37   
38       @Override
39       public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
40           if (request.getSession().getAttribute("user") != null) {
41               serveJSP(request, response, "welcome.jsp");
42           } else {
43               serveJSP(request, response, "register.jsp");
44           }
45       }
46   
47       @Override
48       public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
49           HttpSession session = request.getSession();
50   
51           String email = request.getParameter("email");
52           String name = request.getParameter("firstname") + " " + request.getParameter("lastname");
53           String password = request.getParameter("password");
54           String passwordCheck = request.getParameter("passwordCheck");
55           String phone = request.getParameter("phoneNumber");
56   
57           if (validateRegistration(session, email, password, passwordCheck, phone)) {
58               byte[] salt = hashingService.createSalt();
59               byte[] hashedPassword;
60   
61               try {
62                   hashedPassword = hashingService.hashPassword(salt, password);
63               } catch (InvalidKeySpecException e) {
64                   throw new RuntimeException(e);
65               }
66   
67               User customer = new User(name, email, salt, hashedPassword, User.Role.Customer, phone);
68               userDao.addUser(customer);
69               AccessLog log = new AccessLog(customer.getId(), "registration");
70               logDao.insertLog(log);
71               session.setAttribute("user", customer);
72               serveJSP(request, response, "welcome.jsp");
73           } else {
74               serveJSP(request, response, "register.jsp");
75           }
76       }
77   
78       @Override
79       public void destroy() {
80       }
81   
82       private boolean validateRegistration(HttpSession session, String email, String password, String passwordCheck, String phone) {
83           boolean isValid = true;
84           List<String> errors = new ArrayList<>();
85           if (!email.contains("@")) {
86               errors.add("Email is not valid");
87               isValid = false;
88           }
89           if (userDao.getUserByEmail(email) != null) {
90               errors.add("This email has already been used");
91               isValid = false;
92           }
93           if (!password.equals(passwordCheck)) {
94               errors.add("Passwords do not match");
95               isValid = false;
96           }
97           if (password.length() < 8) {
98               errors.add("Password is less than 8 characters");
99               isValid = false;
100          }
101          if (!phone.matches("\\d{10}")) {
102              errors.add("Phone number can only have digits.");
103              isValid = false;
104          }
105          if (phone.length() != 10) {
106              errors.add("Phone number must be a 10-digit number.");
107              isValid = false;
108          }
109  
110          if (!isValid) {
111              session.setAttribute("errors", errors);
112          }
113          return isValid;
114      }
