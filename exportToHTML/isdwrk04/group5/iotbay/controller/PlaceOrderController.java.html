10   import javax.servlet.http.HttpServletResponse;
11   import java.io.IOException;
12   import java.util.ArrayList;
13   import java.util.List;
14   
15   @WebServlet( name = "placeOrderController", value = "/place-order")
16   public class PlaceOrderController extends BaseServlet {
17   
18       @Override
19       public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
20           Cart cart = (Cart) request.getSession().getAttribute("cart");
21           if (null == cart) {
22               redirectToUrl(request, response, "/cart");
23               return;
24           }
25           serveJSP(request, response, "placeOrder.jsp");
26       }
27   
28       @Override
29       public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
30           User user = (User) request.getSession().getAttribute(USER_ATTRIBUTE);
31   
32           String name = request.getParameter("name");
33           String phone = request.getParameter("phone");
34           String method = request.getParameter("method");
35   
36           if (name.isEmpty() || phone.isEmpty() || method.isEmpty()) {
37               List<String> errors = new ArrayList<>();
38               errors.add("Please enter a name and a phone number.");
39               request.getSession().setAttribute("errors", errors);
40               serveJSP(request, response, "placeOrder.jsp");
41               return;
42           }
43   
44           Order order = new Order();
45           order.setName(name);
46           order.setPhoneNo(phone);
47           order.setMethod(Order.Method.valueOf(method));
48           if (null != user) {
49               order.setUserId(user.getId());
50           }
51           request.getSession().setAttribute("currentOrder", order);
52   
53           // Change this following line here to redirect to shipment or payment controllers
54           redirectToUrl(request, response, "/confirm-order");
55       }
56   }
