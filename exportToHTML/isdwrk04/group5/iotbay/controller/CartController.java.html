1    package isdwrk04.group5.iotbay.controller;
2    
3    import isdwrk04.group5.iotbay.model.Cart;
4    
5    import javax.servlet.ServletException;
6    import javax.servlet.annotation.WebServlet;
7    import javax.servlet.http.HttpServletRequest;
8    import javax.servlet.http.HttpServletResponse;
9    import java.io.IOException;
10   import java.util.Enumeration;
11   
12   @WebServlet( name = "cartController", value = "/cart")
13   public class CartController extends BaseServlet {
14   
15       public final String SAVE_BUTTON_VALUE = "Save changes";
16       public final String CLEAR_BUTTON_VALUE = "Clear cart";
17       public final String ORDER_BUTTON_VALUE = "Place order";
18   
19       @Override
20       public void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException, ServletException {
21           serveJSP(request, response, "cart.jsp");
22       }
23   
24       @Override
25       public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
26           // Action will be null if the remove button is clicked
27           // Setting action to an empty string instead of null prevents the switch statement from throwing a NullPointerException
28           String action = null == request.getParameter("action") ? "" : request.getParameter("action");
29           Cart cart = (Cart) request.getSession().getAttribute("cart");
30           switch (action) {
31               case SAVE_BUTTON_VALUE:
32                   saveOrder(request, response, cart);
33                   break;
34               case CLEAR_BUTTON_VALUE:
35                   clearCart(request, response, cart);
36                   break;
37               case ORDER_BUTTON_VALUE:
38                   placeOrder(request, response, cart);
39                   break;
40               default:
41                   removeItem(request, response, cart);
42                   break;
43           }
44       }
45   
46       private void removeItem(HttpServletRequest request, HttpServletResponse response, Cart cart) throws ServletException, IOException {
47           Enumeration<String> parameters = request.getParameterNames();
48           int index = -1;
49           while (parameters.hasMoreElements()) {
50               String element = parameters.nextElement();
51               if (element.contains("remove")) {
52                   index = Integer.parseInt(element.replace("remove", ""));
53               }
54           }
55           cart.removeProduct(index);
56           request.getSession().setAttribute("cart", cart);
57           serveJSP(request, response, "cart.jsp");
58       }
59   
60       private void placeOrder(HttpServletRequest request, HttpServletResponse response, Cart cart) {
61           updateQuantities(request, cart);
62           request.getSession().setAttribute("cart", cart);
63           redirectToUrl(request, response, "/place-order");
64       }
65   
66       private void clearCart(HttpServletRequest request, HttpServletResponse response, Cart cart) throws ServletException, IOException {
67           cart.clearCart();
68           request.getSession().setAttribute("cart", cart);
69           serveJSP(request, response, "cart.jsp");
70       }
71   
72       private void saveOrder(HttpServletRequest request, HttpServletResponse response, Cart cart) throws ServletException, IOException {
73           updateQuantities(request, cart);
74           request.getSession().setAttribute("cart", cart);
75           serveJSP(request, response, "cart.jsp");
76       }
77   
78       private void updateQuantities(HttpServletRequest request, Cart cart) {
79           int size = cart.getProducts().size();
80           for (int i = 0; i < size; i++) {
81               int quantity = Integer.parseInt(request.getParameter("item" + i));
82               cart.updateQuantity(i, quantity);
83           }
84       }
85   }
